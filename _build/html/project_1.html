#Create a function that assigns the quartile
def assign_quartile(row):
    if row["2022"] <= q1:
        return "Q1"
    elif q1 < row["2022"] <= q2:
        return "Q2"
    elif q2 < row["2022"] <= q3:
        return "Q3"
    else:
        return "Q4"

# Apply the function to the DataFrame

# Calculate the max value in the "2022" column to use as the reference for scaling

# Define a function to create the sparkline scale with '*', using a scale of 10 (10 '*' is the biggest number)
def create_sparkline(value, max_value, scale_factor=10):
    num_stars = int((value / max_value) * scale_factor)
    return "*" * num_stars

# Apply the function to each value in the "2022" column
quartile_data["sparkline"] = quartile_data["2022"].apply(create_sparkline, max_value=max_value)
